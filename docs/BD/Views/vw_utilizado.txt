CREATE VIEW VW_UTILIZADO(UTILIZADO_ID, VALOR_UTILIZADO, PRODUTOR_ID, DISTRIBUIDOR_ID) AS
SELECT GEN_ID(TMP_GEN, 1) AS UTILIZADO_ID,
         COALESCE(SUM(IT.QUANTIDADE * IT.PRECO_UNITARIO),0) AS VALOR_UTILIZADO,
         CN.PRODUTOR_ID,
         CN.DISTRIBUIDOR_ID 
    FROM CAPA_NEGOCIACAO CN,                     
         ITENS_NEGOCIACAO IT                     
   WHERE CN.NEGOCIACAO_ID = IT.NEGOCIACAO_ID     
     AND CN.STATUS IN ('P','A')              
     AND CN.DT_CONCLUSAO IS NULL
  GROUP BY CN.PRODUTOR_ID,
           CN.DISTRIBUIDOR_ID;
