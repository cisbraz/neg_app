CREATE TABLE CAPA_NEGOCIACAO
(
  NEGOCIACAO_ID NUMERIC(8,0) PRIMARY KEY,
  STATUS CHAR(1),
  SOLICITAR_APROVACAO CHAR(1), 
  DT_CADASTRO DATE,
  DT_APROVACAO DATE,
  DT_CONCLUSAO DATE,
  DT_CANCELAMENTO DATE,
  PRODUTOR_ID CHARACTER VARYING(18) REFERENCES PRODUTOR(PRODUTOR_ID),
  DISTRIBUIDOR_ID CHARACTER VARYING(18) REFERENCES DISTRIBUIDOR(DISTRIBUIDOR_ID) 	  
)
